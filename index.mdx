    <Step title="Install the package">

      <CodeGroup>
      ```bash npm
      npm install @kivora/react
      ```
      ```bash yarn
      yarn add @kivora/react
      ```
      ```bash pnpm
      pnpm add @kivora/react
      ```
      </CodeGroup>
    </Step>

    <Step title="Import the styles">
      Import the global stylesheet at the root of your app:
      
      ```tsx
      import '@kivora/react/styles';
      ```
    </Step>

    <Step title="Start building">
      Import and use components in your application:
      
      ```tsx
      import { Button, Badge, Input } from '@kivora/react';

      export default function App() {
        return (
          <div>
            <Button variant="primary" size="md">
              Save changes
            </Button>
            <Badge variant="success">New</Badge>
            <Input 
              label="Email" 
              type="email" 
              placeholder="you@example.com" 
            />
          </div>
        );
      }
      ```
    </Step>
  </Steps>
